价格格式化，补万补亿，逗号分隔

/** 
  * 价格转换，补“万”补“亿”
  * value: 值
  * is_split: 是否用逗号分开
  */
function formatPrice(value, is_split = false){
  let v = value, unit = '';
  if( Math.abs(v) > 10000 & Math.abs(v) < 100000000){
    v = value/10000;
    unit = '万';
  }
  else if( Math.abs(v) > 100000000 ){
    v = value/100000000;
    unit = '亿';
  }

  if(is_split) { return v.toLocaleString() + unit }
  else { return v + unit }
}
