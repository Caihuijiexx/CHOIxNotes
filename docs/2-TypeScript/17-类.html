  • 类的继承

class Person {
  content="Hello";
  sayHello() {
    return content;
  }
}

class Man extends Person {
  sayByebye() {
    return 'Byebye'
  }
}

const Mike = new Man();
console.log(Man.sayHello());  // Hello
console.log(Man.sayByebye()); // Byebye

  • 类的重写

class Person {
  content="Hello";
  sayHello() {
    return content;
  }
}

class Man extends Person {
  sayHello() {
    return 'Hi'
  }
}

const Mike = new Man();
console.log(Man.sayHello());  // Hi

  • super关键字的使用

当新的类还想使用继承类的方法

class Person {
  content="Hello";
  sayHello() {
    return content;
  }
}

class Man extends Person {
  sayHello() {
    return super.sayHello() + "。你好";
  }
}

const Mike = new Man();
console.log(Man.sayHello());  // Hello。你好

  • 访问类型

// public（默认）: 类的内部外部都能使用
// private: 只允许在类的内部使用，局部
// protected: 除了和private一样的限制，允许在继承的子类中使用

class Demo {
  public title: string;
  private data: string;
  protected author: string;
}
class A extends Demo {
  author = this.author;
}

const demo = new Demo()
const a = new A()

console.log(demo.title)
console.log(demo.data)  // 报错
console.log(demo.author)  // 报错
console.log(a.author)

  • 构造函数

class Demo {
    constructor(public name: string){}
}

// 子类继承时，子类的构造函数里面是需要补上super()
// 若父类有需要传的参数也需要补上
class B extends Demo {
  constructor(public age: number){
    super('bb')
  }
}

const demo = new Demo('demo')
const b = new B(18)

console.log(demo.name)    // demo
console.log(b.name)    // bb
console.log(b.age)    // 18
